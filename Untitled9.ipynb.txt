        "import pandas as pd\n",
        "import matplotlib.pyplot as plt\n",
        "import seaborn as sns\n",
        "\n",
        "df=pd.read_csv(\"/content/Average-price-seasonally-adjusted-2022-08.csv\")\n",
        "print(df)\n",
        "\n",
        "# Checking For NAN\n",
        "print(df.isnull().sum())\n",
        "\n",
        "# Checking unique values\n",
        "print(df[\"Region_Name\"].unique())\n",
        "\n",
        "# Converting to numeric\n",

        "df[\"Average_Price_SA\"]=pd.to_numeric(df[\"Average_Price_SA\"],errors=\"raise\" )\n",

        "# Converting to datetime\n",

        "\n",
        "# Filtering Years\n",

        "\n",

        "                )\n",
        "                df_top = yrs[yrs[\"Region_Name\"].isin(top_10_regions)]\n",

        "                # Creating a pivot\n",
        "                df_pivot=df_top.pivot_table(\n",
        "                    index=\"Year\",\n",
        "                        columns=\"Region_Name\",\n",
        "                            values=\"Average_Price_SA\",\n",
        "                                aggfunc=\"mean\"\n",
        "                                )\n",
